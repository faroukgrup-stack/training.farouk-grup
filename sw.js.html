// ملف Service Worker لمنصة إدارة المطاعم - مجموعة فاروق // قم بتغيير رقم الإصدار (v1, v2, ...) بعد كل تعديل لضمان تحديث الكاش const CACHE_NAME = 'fg-restaurants-v3'; // الملفات الأساسية التي سيتم تخزينها مؤقتًا للعمل دون اتصال // يمكنك إضافة المزيد من الملفات حسب الحاجة (CSS، JS، صور) const ASSETS = [ '/', '/index.html', // أضف هنا أي ملفات أساسية أخرى (مثل styles.css، main.js) إذا كانت موجودة ]; // مرحلة التثبيت: تحميل الملفات إلى الكاش self.addEventListener('install', (e) => { e.waitUntil( caches.open(CACHE_NAME).then((cache) => { console.log('تم تثبيت الكاش الجديد لمنصة المطاعم...'); return cache.addAll(ASSETS); }).then(() => { // تفعيل الـ Service Worker الجديد فوراً دون انتظار إغلاق المتصفح return self.skipWaiting(); }) ); }); // مرحلة التنشيط: حذف الكاش القديم self.addEventListener('activate', (e) => { e.waitUntil( caches.keys().then((cacheNames) => { return Promise.all( cacheNames.map((cache) => { if (cache !== CACHE_NAME) { console.log('حذف الكاش القديم:', cache); return caches.delete(cache); } }) ); }).then(() => { // السيطرة على جميع الصفحات المفتوحة فوراً return self.clients.claim(); }) ); }); // استراتيجية الاستجابة: محاولة جلب الملف من الشبكة أولاً، وإذا فشل (عدم اتصال) يتم جلب نسخة من الكاش // هذه الاستراتيجية مناسبة للمواقع التي تتغير بياناتها بشكل متكرر (مثل أدوات الحساب) self.addEventListener('fetch', (e) => { e.respondWith( fetch(e.request).catch(() => { return caches.match(e.request); }) ); });
